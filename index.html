// ── DEVICE DETECTION ──
// Uses multiple signals for accuracy:
// 1. Touch capability
// 2. Screen width
// 3. User agent (as fallback)
// 4. Pointer type (coarse = touch screen)

function isMobileDevice() {
// Pointer media query — most reliable for touch devices
const coarsePointer = window.matchMedia(’(pointer: coarse)’).matches;

// Screen width threshold
const narrowScreen = window.innerWidth <= 768;

// User agent check (fallback)
const mobileUA = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini|Mobile|mobile/i.test(navigator.userAgent);

// Touch points (tablets + phones)
const hasTouch = navigator.maxTouchPoints > 1;

// Mobile = coarse pointer (touch) AND narrow screen, OR strong UA signal
return (coarsePointer && narrowScreen) || (mobileUA && narrowScreen) || (hasTouch && narrowScreen);
}

// Allow manual override via URL param: ?mode=mobile or ?mode=desktop
function getOverride() {
const params = new URLSearchParams(window.location.search);
return params.get(‘mode’); // ‘mobile’ | ‘desktop’ | null
}

// ── ROUTE ──
setTimeout(function() {
const override = getOverride();

if (override === ‘mobile’) {
window.location.replace(‘geolens_mobile.html’);
} else if (override === ‘desktop’) {
window.location.replace(‘geolens_station.html’);
} else if (isMobileDevice()) {
window.location.replace(‘geolens_mobile.html’);
} else {
window.location.replace(‘geolens_station.html’);
}
}, 650); // slight delay for the boot animation
