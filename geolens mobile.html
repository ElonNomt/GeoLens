async function sendMessage(){
  if(isLoading) return;
  const input=document.getElementById('chat-input');
  const q=input.value.trim();
  if(!q) return;
  input.value='';

  const messages=document.getElementById('messages');

  // User bubble
  const userDiv=document.createElement('div');
  userDiv.className='msg-user';
  userDiv.innerHTML=`<div class="msg-user-bubble">${q}</div>`;
  messages.appendChild(userDiv);
  scrollToBottom();

  // AI loading
  const aiDiv=document.createElement('div');
  aiDiv.className='msg-ai';
  aiDiv.innerHTML=`
    <div class="msg-ai-meta">// GEOLENS-79 · PROCESSING <div class="conf-pip loading"></div></div>
    <div class="msg-ai-card">
      <div class="typing-dots"><span></span><span></span><span></span></div>
    </div>`;
  messages.appendChild(aiDiv);
  scrollToBottom();
  isLoading=true;

  try{
    const res=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',
        max_tokens:1000,
        system:`You are GeoLens, an elite geopolitical intelligence analyst. Answer questions about global conflicts, geopolitics, power dynamics, markets, and breaking news with sharp, expert-level clarity. Be concise but substantive — write 2-4 tight paragraphs. No bullet points. No jargon without explanation. End with one short "watch for" sentence highlighting what to monitor next. Keep the tone analytical and authoritative, like a top-tier intelligence brief.`,
        messages:[{role:'user',content:q}]
      })
    });
    const data=await res.json();
    const text=data.content?.map(c=>c.text||'').join('')||'Unable to retrieve analysis. Check connection.';

    // Extract key terms for chips (simple approach: capitalize region/country mentions)
    const chips=extractChips(q+' '+text);

    aiDiv.innerHTML=`
      <div class="msg-ai-meta">// GEOLENS-79 · COMPLETE <div class="conf-pip"></div></div>
      <div class="msg-ai-card">
        <div class="msg-ai-text">${text.replace(/\n\n/g,'<br><br>').replace(/\n/g,'<br>')}</div>
        ${chips.length?`<div class="msg-ai-chips">${chips.map(c=>`<span class="ai-chip">${c}</span>`).join('')}</div>`:''}
      </div>`;
  }catch(err){
    aiDiv.innerHTML=`
      <div class="msg-ai-meta">// GEOLENS-79 · ERROR <div class="conf-pip" style="background:var(--red)"></div></div>
      <div class="msg-ai-card">

  scrollToBottom();
}

function extractChips(text){
  const regions=['China','Russia','Ukraine','Iran','Israel','Gaza','Taiwan','India','Pakistan','Turkey','Saudi Arabia','Yemen','Syria','Niger','Mali','Sudan','Philippines','Japan','Korea','NATO','EU','BRICS'];
  const found=regions.filter(r=>text.toLowerCase().includes(r.toLowerCase()));
  return found.slice(0,5);
}

async function buildMobileMap(){
  if(mMapLoaded) return;
  const container=document.getElementById('mobile-map-container');
  const W=container.clientWidth||390;
  const H=container.clientHeight||360;

  mSvg=d3.select('#mobile-world-svg').attr('width',W).attr('height',H).attr('viewBox',`0 0 ${W} ${H}`);
  mSvg.selectAll('*').remove();

  mProj=d3.geoNaturalEarth1().scale(W/5.8).translate([W/2,H/2]);
  mPath=d3.geoPath().projection(mProj);

  mZoom=d3.zoom().scaleExtent([0.5,8]).on('zoom',(e)=>{
    mTransform=e.transform;
    mG.attr('transform',mTransform);
    rescalePins();
  });

  // Touch-friendly zoom
  mSvg.call(mZoom).on('dblclick.zoom',null);

  mSvg.append('rect').attr('width',W).attr('height',H).attr('fill','#020d06');
  mG=mSvg.append('g');

  // Ocean sphere
  mG.append('path').datum({type:'Sphere'}).attr('d',mPath)
    .attr('fill','#041008').attr('stroke','none');

  // Graticule
  mG.append('path').datum(d3.geoGraticule()())
    .attr('d',mPath).attr('fill','none')
    .attr('stroke','rgba(0,255,136,0.04)').attr('stroke-width',0.5);

  try{
    const world=await d3.json('https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json');

    mG.append('g').selectAll('path')
      .data(topojson.feature(world,world.objects.countries).features)
      .enter().append('path')
      .attr('d',mPath)
      .attr('fill','#0a1f0d')
      .attr('stroke','rgba(0,255,136,0.15)')
      .attr('stroke-width',0.5)
      .on('click',function(event,d){
        // tapped a country — dismiss any open tooltip
        hideMTT();
      });

    mG.append('path')

function drawMobilePins(){

  mapIncidents.forEach(inc=>{

      .attr('class',`mpin mpin-${inc.filter}`)
      .attr('data-id',inc.id)
      .attr('transform',`translate(${x},${y})`)

    // Pulse ring
    const ring=grp.append('circle').attr('r',12).attr('fill','none')

    ring.append('animate').attr('attributeName','r').attr('from',8).attr('to',20)

    ring.append('animate').attr('attributeName','opacity').attr('from',0.4).attr('to',0)

  });
}
